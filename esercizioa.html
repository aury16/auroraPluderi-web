/** * CALCOLATRICE CLI * Uso: node calc.js * Esempio: node calc.js 10 add 5 */ // 1. Estraiamo gli argomenti ignorando i primi due (node e percorso file) const args = process.argv.slice(2); // 2. Controllo se l'utente ha inserito abbastanza argomenti if (args.length < 3) { console.log("Errore: Formato non valido."); console.log("Uso corretto: node calc.js "); process.exit(1); // Chiude lo script con un errore } // 3. Assegniamo i valori a variabili leggibili // Usiamo Number() perché process.argv cattura tutto come stringa const n1 = Number(args[0]); const operazione = args[1].toLowerCase(); // Convertiamo in minuscolo per sicurezza const n2 = Number(args[2]); // 4. Verifichiamo che gli input siano effettivamente numeri if (isNaN(n1) || isNaN(n2)) { console.log("Errore: Inserisci dei numeri validi."); process.exit(1); } let risultato; // 5. Logica della calcolatrice switch (operazione) { case 'add': case '+': risultato = n1 + n2; break; case 'sub': case '-': risultato = n1 - n2; break; case 'mul': case '*': risultato = n1 * n2; break; case 'div': case '/': if (n2 === 0) { console.log("Errore: Divisione per zero impossibile."); process.exit(1); } risultato = n1 / n2; break; default: console.log("Operazione non riconosciuta. Usa: add, sub, mul, div"); process.exit(1); } // 6. Output finale console.log(`-----------------------`); console.log(`Risultato: ${n1} ${operazione} ${n2} = ${risultato}`); console.log(`-----------------------`);
